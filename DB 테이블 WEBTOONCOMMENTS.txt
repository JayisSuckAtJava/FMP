CREATE TABLE WEBTOONCOMMENTS (
COMMENTNUM NUMBER PRIMARY KEY,
ID VARCHAR2(200) NOT NULL,
COMMENTS VARCHAR2(500) NOT NULL,
WRITEDATE DATE NOT NULL,
REC NUMBER NOT NULL,
ARTICLENUM VARCHAR2(200) NOT NULL,
CONSTRAINT "WEBTOONCOMMENTS_ARTICLENUM_FK" FOREIGN KEY (ARTICLENUM)
REFERENCES WEBTOONS(ARTICLENUM)
);
